Struct std::prelude::Option

Re-export from std::option::Option

struct Option<T> { ... }

source

Optional value

Optional values can either contain a value or they can be empty. See also Result.

Example

fn divide(a: i32, b: i32) -> Option<i32> {
    if b == 0 {
        Option::none()
    } else {
        Option::some(a / b)
    }
}

assert_eq!(divide(10, 5), Option::some(2));
assert_eq!(divide(10, 0), Option::none());
