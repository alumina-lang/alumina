Macro std::fmt::format

macro format($fmt_str, $arg...) { ... }

source

Format a string into a StringBuf.

Example

use std::fmt::format;
use std::string::StringBuf;

let s: StringBuf = format!("{} + {} = {}", 1, 2, 3).unwrap();
defer s.free();

assert_eq!(s[..] as &[u8], "1 + 2 = 3");
