Struct std::option::Option

struct Option<T> { ... }

source

Optional value

Optional values can either contain a value or they can be empty.

See also module-level documentation.

Example

let a = Option::some(1);
let b: Option<i32> = Option::none();

assert_eq!(a.is_some(), true);
assert_eq!(b.is_some(), false);
