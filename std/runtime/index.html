Functions

  • fn in_const_context() -> bool

    source

    Returns true when evaluated in a const context and false otherwise.

    Example

    use std::runtime::in_const_context;

const IN_CONST_CONTEXT: bool = in_const_context();

assert!(IN_CONST_CONTEXT);
assert!(!in_const_context());
