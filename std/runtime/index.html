  • fn in_coroutine() -> bool

    source

    Returns true if the current code is running in a coroutine.

    Example

    use std::runtime::in_coroutine;

fn* gen() -> Coroutine<bool> {
    yield in_coroutine();
}

let gen = gen();
defer gen.free();

assert!(!in_coroutine());
assert!(gen.next().unwrap());
