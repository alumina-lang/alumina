Struct std::thread::pool::ThreadPool

struct ThreadPool { ... }

source

A thread pool.

Example

use std::thread::pool::ThreadPool;
use std::sync::Atomic;

// Use 10 threads
let pool = ThreadPool::unbounded(10);
let counter = sync::Atomic::new(0);

for i in 0..1000 {
    pool
        .spawn_detached(|&counter| {
            counter.fetch_add(1, sync::Ordering::Relaxed);
        })
        .unwrap();
}

// Wait for all remaining tasks to complete
pool.join();

assert_eq!(counter.load(sync::Ordering::Relaxed), 1000);

    max_in_flight is the maximum length of the task queue, if the queue is full, calling spawn will block.
