Struct std::fs::File

Re-export from std::fs::unix::File

struct File { ... }

source

A file.

Example

use std::fs::{File, Path};
use std::string::StringBuf;

let file = File::open(Path::new("/etc/passwd")).unwrap();
defer file.close();

let buf: StringBuf = StringBuf::new();
defer buf.free();

file.read_to_end(&buf).unwrap();

println!("{}", buf[..]);
