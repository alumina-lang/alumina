  • fn env() -> EnvVars

    source

    Return environment variables of the current process.

    Example

    use std::process::env;
use std::string::split;

let path_dirs = env()
    .get("PATH")
    .unwrap()
    .split(":");

for dir in path_dirs {
    println!("{}", dir);
}
