  • fn env() -> EnvIterator

    source

    Return environment variables of the current process.

    Example

    use std::process::env;
use std::string::split;

let path_dirs = env()
    .filter(|e: (&[u8], &[u8])| -> bool { e.0 == "PATH" })
    .next()
    .unwrap().1
    .split(":");

for dir in path_dirs {
    println!("{}", dir);
}
