  • fn exit(code: i32) -> !

    source

    Exits the process with a given return code.

    This is a graceful exit (atexit handlers will run). For an immediate abort, see _exit or abort.

    Example

    use std::process::exit; 

exit(0); // Exit with success
exit(1); // Exit with error

  • fn env() -> EnvVars

    source

    Return environment variables of the current process.

    Example

    use std::process::env;
use std::string::split;

let path_dirs = env()
    .get("PATH")
    .unwrap()
    .split(":");

for dir in path_dirs {
    println!("{}", dir);
}
